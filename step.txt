sudo usermod -aG sudo kolla
sudo visudo

sudo python3 -m venv /kolla-env

source /kolla-env/bin/activate

pip install -U pip

sudo mkdir -p /etc/kolla
sudo chown $USER:$USER /etc/kolla

pip install git+https://opendev.org/openstack/kolla-ansible@unmaintained/2023.1

cp -r /kolla-env/share/kolla-ansible/etc_examples/kolla/* /etc/kolla
